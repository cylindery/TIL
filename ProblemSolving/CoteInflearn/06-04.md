```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());
        int s = Integer.parseInt(st.nextToken());
        int n = Integer.parseInt(st.nextToken());
        st = new StringTokenizer(br.readLine());
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = Integer.parseInt(st.nextToken());

        System.out.println(solution(s, n, arr));
        br.close();
    }

    public static String solution(int s, int n, int[] arr) {
        StringBuilder sb = new StringBuilder();
        Queue<Integer> q = new LinkedList<>();

        for (int a : arr) {
            if (!q.contains(a)) {
                q.offer(a);
                if (q.size() > s) q.poll();
            } else {
                int cnt = 0;
                while (q.peek() != a) {
                    q.offer(q.poll());
                    cnt++;
                }
                q.poll();
                for (int i = 0; i < q.size() - cnt; i++) q.offer(q.poll());
                q.offer(a);
            }
        }

        ArrayList<Integer> list = new ArrayList<>(q);
        for (int i = list.size() - 1; i >= 0; i--) sb.append(list.get(i)).append(" ");

        return sb.toString();
    }
}
```

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());
        int s = Integer.parseInt(st.nextToken());
        int n = Integer.parseInt(st.nextToken());
        st = new StringTokenizer(br.readLine());
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = Integer.parseInt(st.nextToken());

        System.out.println(solution(s, n, arr));
        br.close();
    }

    public static String solution(int s, int n, int[] arr) {
        StringBuilder sb = new StringBuilder();
        int[] cache = new int[s];

        for (int a : arr) {
            boolean flag = false;
            int idx = 0;
            for (int i = 0; i < s; i++) {
                if (cache[i] == a) {
                    flag = true;
                    idx = i;
                    break;
                }
            }
            if (!flag) idx = s - 1;
            for (; idx >= 1; idx--) {
                cache[idx] = cache[idx - 1];
            }
            cache[0] = a;
        }
        for (int c : cache) sb.append(c).append(" ");

        return sb.toString();
    }
}
```

### 참고한 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());
        int s = Integer.parseInt(st.nextToken());
        int n = Integer.parseInt(st.nextToken());
        st = new StringTokenizer(br.readLine());
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = Integer.parseInt(st.nextToken());

        System.out.println(solution(s, n, arr));
        br.close();
    }

    public static String solution(int s, int n, int[] arr) {
        StringBuilder sb = new StringBuilder();
        int[] cache = new int[s];

        for (int a : arr) {
            int virus = -1;
            for (int i = 0; i < s; i++) if (cache[i] == a) virus = i;
            if (virus == -1) virus = s - 1;
            for (int i = virus; i >= 1; i--) {
                cache[i] = cache[i - 1];
            }
            cache[0] = a;
        }
        for (int c : cache) sb.append(c).append(" ");

        return sb.toString();
    }
}
```

### 피드백

- 이전의 삽입 정렬을 이용해서 LRU 알고리즘 구현하기.
- LRU 알고리즘은 오랫동안 참조되지 않은 페이지를 교체하는 기법이다.
    - 이를 cache라는 배열을 구현한 뒤, 새로운 작업에 대해 캐시 안에 있는 작업이면 그 작업의 인덱스부터, 그렇지 않으면 제일 끝의 인덱스부터 삽입 정렬하는 것과 같다.
    - 그리고 항상 지금 들어온 작업을 최신 인덱스, 즉 0에 삽입하는 방법.
