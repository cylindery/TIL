# 무인도 여행

> https://school.programmers.co.kr/learn/courses/30/lessons/154540

### 내 코드

```java
import java.util.*;

class Solution {
    int n, m;
    int[][] map;
    int[] dr = {0, 0, 1, -1};
    int[] dc = {1, -1, 0, 0,};
    ArrayList<Integer> ans;
    
    public int[] solution(String[] maps) {
        ans = new ArrayList<>();
        
        n = maps.length;
        m = maps[0].length();
        map = new int[n][m];
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                char ch = maps[i].charAt(j);
                if (ch == 'X') {
                    map[i][j] = 0;
                } else {
                    map[i][j] = ch - '0';
                }
            }
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (map[i][j] > 0) {
                    bfs(i, j);
                }
            }
        }
        
        if (ans.size() == 0) {
            ans.add(-1);
        }
        
        int[] arr = new int[ans.size()];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = ans.get(i);
        }
        
        Arrays.sort(arr);
        
        return arr;
    }
    
    public void bfs(int r, int c) {
        Queue<Pos> q = new ArrayDeque<>();
        q.offer(new Pos(r, c));
        int sum = map[r][c];
        map[r][c] = 0;
        
        while (!q.isEmpty()) {
            Pos cur = q.poll();
            
            for (int i = 0; i < 4; i++) {
                int nr = cur.r + dr[i];
                int nc = cur.c + dc[i];
                if (nr < 0 || nc < 0 || nr >= n || nc >= m) continue;
                if (map[nr][nc] > 0) {
                    q.offer(new Pos(nr, nc));
                    sum += map[nr][nc];
                    map[nr][nc] = 0;
                }
            }
        }
        
        ans.add(sum);
    }
    
    static class Pos {
        int r, c;
        
        public Pos(int r, int c) {
            this.r = r;
            this.c = c;
        }
    }
}
```
