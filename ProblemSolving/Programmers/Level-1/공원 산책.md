# 공원 산책

> https://school.programmers.co.kr/learn/courses/30/lessons/172928

### 내 코드

```java
import java.util.*;

class Solution {
    static int n, m;
    static boolean[][] map;
    
    public int[] solution(String[] park, String[] routes) {
        n = park.length;
        m = park[0].length();
        map = new boolean[n][m];
        int r = 0, c = 0;
        
        // 맵 입력
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (park[i].charAt(j) == 'S') {
                    r = i;
                    c = j;
                } else if (park[i].charAt(j) == 'X') {
                    map[i][j] = true;
                }
            }
        }
        
        for (String route : routes) {
            String dir = route.split(" ")[0];
            int cnt = Integer.parseInt(route.split(" ")[1]);
            
            int nr = r;
            int nc = c;
            for (int i = 0; i < cnt; i++) {
                if (dir.equals("E")) nc++;
                else if (dir.equals("W")) nc--;
                else if (dir.equals("S")) nr++;
                else nr--;
                
                if (nr < 0 || nc < 0 || nr >= n || nc >= m) break;
                if (map[nr][nc]) break;
                if (i == cnt - 1) {
                    r = nr;
                    c = nc;
                }
            }
        }
        
        int[] ans = {r, c};
        return ans;
    }
}
```
