# 종이의 개수

> https://www.acmicpc.net/problem/1780

### 내 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {

    private static int[][] paper; // N*N 행렬 종이
    private static int minus = 0, zero = 0, plus = 0; // 각각 -1, 0, 1로만 채워진 종이의 개수

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();

        int N = Integer.parseInt(br.readLine());

        paper = new int[N][N];
        for (int i = 0; i < N; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < N; j++) {
                paper[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        checkPaper(0, 0, N);

        sb.append(minus).append("\n").append(zero).append("\n").append(plus);
        System.out.println(sb);

        br.close();
    }

    // {row, column}부터 N*N 크기의 종이에 입력된 수 검사
    private static void checkPaper(int row, int column, int N) {
        int target = paper[row][column]; // 첫 항 기준

        // 더이상 9등분할 수 없으면 그 원소 카운팅한 뒤 메서드 종료
        if (N == 1) {
            count(target);
            return;
        }

        // N*N 루프하며 하나라도 다르면 루프 종료
        boolean flag = true;
        outer:
        for (int i = row; i < row + N; i++) {
            for (int j = column; j < column + N; j++) {
                if (paper[i][j] != target) {
                    flag = false;
                    break outer;
                }
            }
        }

        // true: 카운팅, false: 9등분한 뒤 재귀
        if (flag) {
            count(target);
        } else {
            int block = N / 3;
            for (int i = row; i < row + N; i += block) {
                for (int j = column; j < column + N; j += block) {
                    checkPaper(i, j, block);
                }
            }
        }
    }

    // -1, 0, 1 중 하나로만 채워진 종이 카운팅
    private static void count(int target) {
        switch (target) {
            case -1:
                minus++;
                break;
            case 0:
                zero++;
                break;
            case 1:
                plus++;
                break;
        }
    }

}
```

### 내 코드 2

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {

    static int n, m = 0, z = 0, o = 0;
    static int[][] map;
    static boolean[][] vis;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());
        map = new int[n][n];
        for (int i = 0; i < n; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < n; j++) {
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        vis = new boolean[n][n];

        int len = n;
        while (len != 0) {
            for (int i = 0; i < n; i += len) {
                for (int j = 0; j < n; j += len) {
                    check(i, j, len);
                }
            }
            len /= 3;
        }

        System.out.println(m + "\n" + z + "\n" + o);
        br.close();
    }

    public static void check(int r, int c, int len) {
        int s = map[r][c];
        boolean flag = true;

        outer:
        for (int i = r; i < r + len; i++) {
            for (int j = c; j < c + len; j++) {
                if (map[i][j] != s || vis[i][j]) {
                    flag = false;
                    break outer;
                }
            }
        }

        if (flag) {
            for (int i = r; i < r + len; i++) {
                for (int j = c; j < c + len; j++) {
                    vis[i][j] = true;
                }
            }

            if (s == -1) m++;
            else if (s == 0) z++;
            else o++;
        }
    }
}
```

### 피드백

- 분할 정복을 제대로 풀어본 두 번째 문제. 전에 풀었던 복잡한 별 찍기와 유사한 방법으로 풀었다.
    - 다른 방법으로는 쿼드 트리를 이용한 방법이 있다고 하는데, 좀 더 공부해봐야겠다.
- 분할 정복과 재귀를 온전히 내 힘으로 풀어낸 것 같아 기분이 좋다. 너무 어려워하지 말고 자신감있게 도전하자.
- 시간이 흘러 다시 풀어보았다. 문제의 목표인 분할 정복과 재귀로 풀지 않았다... 그 의도에 맞게 푸는 것이 훈련에는 더 도움될 것 같다.
