# 색종이 만들기

> https://www.acmicpc.net/problem/2630

### 내 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {

    private static int[][] paper; // 전체 종이
    private static int white = 0, blue = 0; // 하얀색, 파란색 색종이 개수

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        int N = Integer.parseInt(br.readLine()); // 전체 종이 한 변의 길이

        // 전체 종이 입력
        paper = new int[N][N];
        for (int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());
            for (int j = 0; j < N; j++) {
                paper[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        // 하얀색, 파란색 색종이 개수 계산
        cutPaper(0, 0, N);

        System.out.println(white + "\n" + blue);

        br.close();
    }

    private static void cutPaper(int row, int col, int N) {
        int target = paper[row][col];

        // 더이상 4등분할 수 없으면 종료
        if (N == 1) {
            count(target);
            return;
        }

        // N*N 루프하며 하나라도 다르면 루프 종료
        boolean flag = true;
        outer:
        for (int i = row; i < row + N; i++) {
            for (int j = col; j < col + N; j++) {
                if (paper[i][j] != target) {
                    flag = false;
                    break outer;
                }
            }
        }

        // true: 카운팅, false: 4등분한 뒤 재귀
        if (flag) {
            count(target);
        } else {
            int block = N / 2;
            for (int i = row; i < row + N; i+= block) {
                for (int j = col; j < col + N; j += block) {
                    cutPaper(i, j, block);
                }
            }
        }
    }

    // 0 or 1 종이 카운팅
    private static void count(int target) {
        if (target == 0) {
            white++;
        } else {
            blue++;
        }
    }

}
```
