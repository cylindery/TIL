# 그림 교환

> https://www.acmicpc.net/problem/1029

### 내 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Main {

    private static int n, ans;
    private static int[][] cost;
    private static boolean[] visited;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine());
        ans = 1;

        cost = new int[n][n];
        for (int i = 0; i < n; i++) {
            String str = br.readLine();
            for (int j = 0; j < n; j++) {
                cost[i][j] = str.charAt(j) - '0';
            }
        }

        visited = new boolean[n];
        visited[0] = true;

        dfs(0, 0, 1);
        System.out.println(ans);
        br.close();
    }

    private static void dfs(int cur, int prevCost, int cnt) {
        // 더이상 갈 수 있는 곳이 없으면 종료
        if (!isValid(cur, prevCost)) {
            ans = Math.max(ans, cnt);
            return;
        }

        // 현재 가능한 곳들 dfs 체크하면서 돌리기
        for (int i = 0; i < n; i++) {
            if (i== cur) continue;
            if (!visited[i] && cost[cur][i] >= prevCost) {
                visited[i] = true;
                dfs(i, cost[cur][i], cnt + 1);
                visited[i] = false;
            }
        }
    }

    private static boolean isValid(int cur, int prevCost) {
        for (int i = 0; i < n; i++) {
            if (i == cur) continue;
            if (!visited[i] && cost[cur][i] >= prevCost) {
                return true;
            }
        }
        return false;
    }
}
```

### 참고한 내 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;

public class Main {

    private static int n, ans;
    private static int[][] cost, dp;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine());
        ans = 1;

        cost = new int[n][n];
        for (int i = 0; i < n; i++) {
            String str = br.readLine();
            for (int j = 0; j < n; j++) {
                cost[i][j] = str.charAt(j) - '0';
            }
        }

        dp = new int[n][1 << n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dp[i], -1);
        }

        dfs(0, 1, 1);
        System.out.println(ans);
        br.close();
    }

    private static void dfs(int cur, int visited, int cnt) {
        ans = Math.max(ans, cnt);

        if (ans == n || visited == (1 << n) - 1) {
            return;
        }

        for (int next = 0; next < n; next++) {
            if ((visited & (1 << next)) == 0 && dp[cur][visited] <= cost[cur][next]) {
                if (dp[next][visited | (1 << next)] != -1 && dp[next][visited | (1 << next)] <= cost[cur][next]) continue;
                dp[next][visited | (1 << next)] = cost[cur][next];
                dfs(next, visited | (1 << next), cnt + 1);
            }
        }
    }
}
```
