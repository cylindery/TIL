### 내 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.StringTokenizer;

public class Main {

    private static int N; // 과제 수 1~1000
    private static boolean[][] assignments; // {마감일, 과제 점수}

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        N = Integer.parseInt(br.readLine());

        assignments = new boolean[1001][101]; // 과제 입력
        int dayMax = 0; // 마감일 최댓값
        StringTokenizer st;
        while (N-- > 0) {
            st = new StringTokenizer(br.readLine(), " ");
            int d = Integer.parseInt(st.nextToken());
            int w = Integer.parseInt(st.nextToken());
            assignments[d][w] = true;
            dayMax = Math.max(dayMax, d);
        }

        // 마지막날부터 역으로 선택가능한 최대 과제 점수 더하고, false
        int result = 0;
        for (int i = dayMax; i >= 1; i--) {
            int temp = 0;

            for (int j = i; j <= dayMax; j++) {
                temp = Math.max(getMax(j), temp);
            }

            assignments[index][temp] = false;


        }
    }

    // 해당 인덱스에서 최댓값 점수 구하기
    private static int getMax(int index) {
        int max = -1;

        for (int i = 0; i <= 100; i++) {
            if (assignments[index][i] && i > max) {
                max = i;
            }
        }

        return max;
    }
}
```
