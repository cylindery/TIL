# 해킹

> https://www.acmicpc.net/problem/10282

### 내 코드

```java
import java.io.*;
import java.util.*;

public class Main {

    static int n, pcCnt, timeCnt;
    static ArrayList<ArrayList<Integer>> list;
    static int[][] time;
    static int[] inDegree;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringTokenizer st;

        int t = Integer.parseInt(br.readLine());
        while (t-- > 0) {
            st = new StringTokenizer(br.readLine());
            n = Integer.parseInt(st.nextToken());
            int d = Integer.parseInt(st.nextToken());
            int c = Integer.parseInt(st.nextToken());

            // 컴퓨터 별 다음 컴퓨터 순서 리스트
            list = new ArrayList<>();
            for (int i = 0; i < n + 1; i++) {
                list.add(new ArrayList<>());
            }

            // 의존성, 감염 시간 입력. 들어오는 간선 수 입력
            time = new int[n + 1][n + 1];
            inDegree = new int[n + 1];
            for (int i = 0; i < d; i++) {
                st = new StringTokenizer(br.readLine());
                int a = Integer.parseInt(st.nextToken());
                int b = Integer.parseInt(st.nextToken());
                int s = Integer.parseInt(st.nextToken());
                list.get(b).add(a);
                time[b][a] = s;
                inDegree[a]++;
            }

            topologicalSort(c);
            bw.write(pcCnt + " " + timeCnt + "\n");
        }

        bw.flush();
        bw.close();
        br.close();
    }

    static void topologicalSort(int start) {
        Queue<Integer> queue = new LinkedList<>();
        Integer[] sum = new Integer[n + 1];

        // 각 노드까지 걸리는 시간 초기화. 큐에 시작점 삽입
        sum[start] = 0;
        queue.offer(start);

        pcCnt = 1;
        timeCnt = 0;
        while (!queue.isEmpty()) {
            int now = queue.poll();

            for (int next : list.get(now)) {
                if (sum[next] == null) {
                    sum[next] = time[now][next];
                } else {
                    sum[next] = Math.min(sum[next], sum[now] + time[now][next]);
                }
                inDegree[next]--;

                if (inDegree[next] == 0) {
                    queue.offer(next);
                    pcCnt++;
                }
            }
        }

        for (Integer val : sum) {
            if (val != null) {
                timeCnt = Math.max(timeCnt, val);
            }
        }
    }

}
```
